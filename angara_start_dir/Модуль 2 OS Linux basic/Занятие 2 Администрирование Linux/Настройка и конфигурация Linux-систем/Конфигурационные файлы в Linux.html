Конфигурационные файлы в Linux

Цель работы:

  • Разобраться какой конфигурационный файл за что отвечает;
  • Применить команды - предоставить чёткое описание, что делает каждая команда в конкретном конфигурационном файле.
  • Составить отчёт с описанием и скриншотами.

Защита:

  • Ответить на вопросы;
  • Продемонстрировать конфигурацию фалов.

На самом деле в самой системы Linux конфигурационных файлов нет. Поскольку операционная система - это всего лишь набор программ и ядро, то все эти файлы были созданы определенными программами и читаются ими же для настройки поведения. Большинство файлов, которые мы привыкли считать стандартными, относятся к системе инициализации или к другим системным утилитам.

Конфигурационные файлы Linux

  1. /etc/adjtime
    1. sudo cat /etc/adjtime
    1. sudo hwclock --systohc
  1. /etc/bash.bashrc
    1. sudo echo "export CUSTOM_VAR='value'" >> /etc/bash.bashrc
    1. source /etc/bash.bashrc
  1. /etc/crontab
    1. sudo nano /etc/crontab
    1. sudo crontab -l
  1. /etc/environment
    1. sudo echo "NEW_VARIABLE=value" >> /etc/environment
    1. sudo cat /etc/environment
  1. /etc/fstab
    1. sudo nano /etc/fstab
    1. sudo mount -a
  1. /etc/group
    1. sudo cat /etc/group
    1. sudo usermod -aG groupname username
  1. /etc/hostname
    1. sudo cat /etc/hostname
    1. sudo hostnamectl set-hostname newhostname
  1. /etc/hosts
    1. sudo sh -c 'echo "127.0.0.2 example.com" >> /etc/hosts'
    1. sudo sed -i '/example.com/d' /etc/hosts
  1. /etc/hosts.allow и /etc/hosts.deny
    1. sudo sh -c 'echo "sshd: 192.168.1.0/24" >> /etc/hosts.allow’
    1. sudo sh -c 'echo "ALL: ALL" >> /etc/hosts.deny’
  1. /etc/issue и /etc/issue.net
    1. sudo sh -c 'echo "Welcome to our server" > /etc/issue’
    1. sudo sh -c 'echo "Authorized access only" > /etc/issue.net’
  1. /etc/ld.so.conf
    1. sudo sh -c 'echo "/usr/local/lib" >> /etc/ld.so.conf’
    1. sudo sed -i '/\/usr\/local\/lib/d' /etc/ld.so.conf
  1. /etc/localtime
    1. sudo ln -sf /usr/share/zoneinfo/Zone/SubZone /etc/localtime
  1. /etc/login.defs
    1. sudo sed -i 's/^PASS_MAX_DAYS.*/PASS_MAX_DAYS 90/' /etc/login.defs
  1. /etc/mime.types
    1. sudo sh -c 'echo "application/x-custom-type custom" >> /etc/mime.types’
    1. sudo sed -i '/^application\/x-custom-type/d' /etc/mime.types
  1. /etc/modprobe.d/
    1. sudo touch /etc/modprobe.d/custom.conf
  1. /etc/modules-load.d/
    1. sudo touch /etc/modules-load.d/custom.conf
  1. /etc/nsswitch.conf
    1. sudo sed -i 's/^passwd:.*$/passwd: files ldap/' /etc/nsswitch.conf
  1. /etc/ntp.conf
    1. sudo sh -c 'echo "server ntp.example.com" >> /etc/ntp.conf'
  1. /etc/os-release
    1. sudo sed -i 's/^VERSION_ID=.*/VERSION_ID="20.04"/' /etc/os-release
  1. /etc/passwd
    1. sudo useradd -m newuser
  1. /etc/motd
    1. sudo nano /etc/motd
  1. /etc/profile
    1. sudo sh -c 'echo "export CUSTOM_VAR=value" >> /etc/profile’
  1. /etc/resolv.conf
    1. sudo sh -c 'echo "nameserver 8.8.8.8" > /etc/resolv.conf’
  1. /etc/sddm.conf
    1. sudo cp /usr/share/sddm/scripts/Xsetup /etc/sddm.conf
  1. /etc/shadow
    1. sudo useradd -m newuser
      sudo passwd newuser
  1. /etc/sudoers
    1. sudo sh -c 'echo "username ALL=(ALL) NOPASSWD:ALL" >> /etc/sudoers’
  1. /etc/sysctl.conf
    1. sudo sh -c 'echo "kernel.sysrq = 0" >> /etc/sysctl.conf’
  1. /etc/vconsole.conf
    1. sudo sh -c 'echo "KEYMAP=us" > /etc/vconsole.conf’
  1. /etc/services
    1. sudo sh -c 'echo "newservice 12345/tcp" >> /etc/services’
  1. /boot/grub/grub.cfg
    1. этот файл лучше не редактировать напрямую, поскольку он может быть перезаписан при обновлении GRUB. Вместо этого, настройки GRUB лучше изменять через файлы конфигурации в /etc/default/grub - sudo sed -i 's/GRUB_DEFAULT=0/GRUB_DEFAULT=saved/' /etc/default/grub, sudo sed -i 's/#GRUB_TERMINAL_OUTPUT=console/GRUB_TERMINAL_OUTPUT=console/' /etc/default/grub, sudo update-grub
